Replace ids with names in the occurs table:
SELECT m.name, s.name, w.name, o.dTime
FROM occurs o
JOIN managers m ON o.mid = m.id
JOIN students s ON o.sid = s.id
JOIN writeUps w ON o.wid = w.id
ORDER BY o.dTime

Number of times managers have used certain types of write ups:
SELECT m.name, w.name, COUNT(*)
FROM occurs o
JOIN managers m ON o.mid = m.id
JOIN writeUps w ON o.wid = w.id
GROUP BY m.name, w.name
ORDER BY w.name

Number of times mangers have writen up certain students:
SELECT m.name, s.name, COUNT(*)
FROM occurs o
JOIN managers m ON o.mid = m.id
JOIN students s ON o.sid = s.id
GROUP BY m.name, s.name

Number of times students have received certain types of write ups:

FROM occurs o
JOIN students s ON o.sid = s.id
JOIN writeUps w ON o.wid = w.id
GROUP BY s.name, w.name

Get all write ups for employees that work at the same time and on the same day:
SELECT m.name, s.name, wr.name, o.dTime, d.name, o.strt
FROM (SELECT o.mid, w.sid, o.wid, o.dTime, w.did, w.strt
      FROM occurs o
      RIGHT JOIN (SELECT w1.sid, w1.did, w1.strt
                  FROM works w1
                  JOIN works w2 ON w1.did = w2.did AND w1.strt = w2.strt AND NOT w1.sid = w2.sid) w
      ON o.sid = w.sid) o
LEFT JOIN managers m ON o.mid = m.id
LEFT JOIN students s ON o.sid = s.id
LEFT JOIN writeUps wr ON o.wid = wr.id
LEFT JOIN days d ON o.did = d.id
ORDER BY o.dTime

Get all write ups for employees that work at the same area on the same day:
SELECT m.name, s.name, wr.name, o.dTime, d.name, a.name
FROM (SELECT o.mid, w.sid, o.wid, o.dTime, w.did, w.aid
      FROM occurs o
      RIGHT JOIN (SELECT w1.sid, w1.did, w1.aid
                  FROM works w1
                  JOIN works w2 ON w1.aid = w2.aid AND w1.did = w2.did AND NOT w1.sid = w2.sid) w
      ON o.sid = w.sid) o
LEFT JOIN managers m ON o.mid = m.id
LEFT JOIN students s ON o.sid = s.id
LEFT JOIN writeUps wr ON o.wid = wr.id
LEFT JOIN days d ON o.did = d.id
LEFT JOIN areas a ON o.aid = a.id
ORDER BY o.dTime
